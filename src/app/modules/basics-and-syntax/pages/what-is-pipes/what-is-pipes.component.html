Just standart string. Syntax on double braces {{now}}

With pipe. Syntax: "now | date" {{ now | date }}

With pipe. Syntax: "now | date: 'long'" {{ now | date: 'long' }}

With pipe. Syntax: "now | date:'long' | uppercase" {{ now | date:'long' | uppercase }}

With pipe. Syntax: "now | date:'long' | lowercase" {{ now | date:'long' | lowercase }}
